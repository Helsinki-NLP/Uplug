001    package org.maltparser.core.propagation.spec;
002    
003    import java.util.ArrayList;
004    
005    
006    /**
007     * @author Johan Hall
008     *
009     */
010    public class PropagationSpecs extends ArrayList<PropagationSpec> {
011            public static final long serialVersionUID = 1L;
012            
013            public PropagationSpecs() {
014                    super();
015            }
016    
017            @Override
018            public String toString() {
019                    StringBuilder sb = new StringBuilder();
020                    for (PropagationSpec spec: this) {
021                            sb.append(spec.toString() + "\n");
022                    }
023                    
024                    return sb.toString();
025            }
026    }
